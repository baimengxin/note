```ts
Page({
  data: {
    brands: [], // 后端返回的所有品牌数据
    displayBrands: [], // 要显示的品牌（最多6个）
    selectedBrands: {}, // 选中状态的管理对象，key 为品牌 id，value 为布尔值
  },

  onLoad() {
    this.fetchBrands(); // 在页面加载时获取品牌数据
  },

  // 从后端获取品牌数据并初始化
  fetchBrands() {
    // 假设这里是调用后端 API 获取数据
    wx.request({
      url: "https://your-api-url/brands", // 替换为你的 API 地址
      method: "GET",
      success: (res) => {
        const brands = res.data;

        // 初始化选中状态对象，默认所有品牌未选中
        const selectedBrands = {};
        brands.forEach((brand) => {
          selectedBrands[brand.id] = false; // 假设品牌对象有一个唯一的 id 属性
        });

        this.setData(
          {
            brands,
            selectedBrands,
          },
          () => {
            this.updateDisplayBrands(); // 根据新数据更新要显示的品牌
          }
        );
      },
      fail: (err) => {
        console.error("Failed to fetch brands", err);
      },
    });
  },

  // 更新显示品牌列表
  updateDisplayBrands() {
    const { brands } = this.data;
    const displayBrands = brands.length > 6 ? brands.slice(0, 6) : brands;
    this.setData({ displayBrands });
  },

  // 切换品牌选中状态
  toggleSelectBrand(e) {
    const index = e.currentTarget.dataset.index;
    const brand = this.data.displayBrands[index]; // 获取点击的品牌数据
    const brandId = brand.id; // 假设品牌数据中有 id 字段

    // 更新选中状态
    this.setData({
      [`selectedBrands.${brandId}`]: !this.data.selectedBrands[brandId],
    });
  },

  // 跳转到品牌页
  navigateToBrandPage() {
    wx.navigateTo({
      url: "/pages/brandPage/brandPage", // 跳转到品牌页的路径
    });
  },

  // 将选中的品牌状态传递给后端
  applyFilters() {
    const { selectedBrands } = this.data;

    // 获取所有选中的品牌 ID
    const selectedBrandIds = Object.keys(selectedBrands).filter(
      (id) => selectedBrands[id]
    );

    // 发起请求，将选中的品牌 ID 传递给后端
    wx.request({
      url: "https://your-api-url/apply-filters", // 替换为你的筛选 API 地址
      method: "POST",
      data: {
        brandIds: selectedBrandIds,
      },
      success: (res) => {
        console.log("Filters applied successfully", res.data);
      },
      fail: (err) => {
        console.error("Failed to apply filters", err);
      },
    });
  },
});
```

```html
<view class="brand-list">
  <block wx:for="{{displayBrands}}" wx:key="index">
    <!-- 根据品牌的选中状态应用不同的类名 -->
    <view
      class="brand-item {{selectedBrands[item.id] ? 'selected' : ''}}"
      data-index="{{index}}"
      bindtap="toggleSelectBrand"
    >
      {{item.name}}
    </view>
  </block>

  <!-- 如果品牌数量超过6个，显示“更多”按钮 -->
  <view
    wx:if="{{brands.length > 6}}"
    class="more-button"
    bindtap="navigateToBrandPage"
    >更多</view
  >
</view>

<!-- 应用筛选按钮 -->
<button bindtap="applyFilters">应用筛选</button>
```
